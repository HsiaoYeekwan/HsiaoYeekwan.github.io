## 题目传送门 ：[CodeForces - 404C](http://codeforces.com/problemset/problem/404/C)
>题意
构造一个n阶图，每个点（编号1-n）的度数不超过k。
第二行n个数为顶点到节点i的最短距离。

```c
#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <string>
#include <algorithm>
#include <queue>
#include <vector>
#include <set>
#define inf 99999999

using namespace std;
typedef long long ll;
const int maxn = 1e5+10;
int n,k;
struct node{
    int v,id;
    node(int vv = 0,int idd = 0):v(vv),id(idd){}
};
node a[maxn];
int d[maxn];
vector<node> ans;
int pos[maxn];//pos数组存储最短距离为i的区间，即（pos[i-1],pos[i] ];

bool cmp(const node &a,const node &b){
    return a.v < b.v;
}

int main(int argc, char const *argv[])
{
    while(cin >> n >> k){
        memset(pos,0,sizeof(pos));
        memset(d,0,sizeof(d));
        ans.clear();
        for(int i = 1 ; i <= n ; i ++){
            scanf("%d",&a[i].v);
            a[i].id = i;
        }
        sort(a+1,a+1+n,cmp);
        if(a[2].v != 1){cout << "-1" << endl;continue;}
        int Max = a[n].v;
        int now = 1;
        for(int i = 1 ; i <= n ; i ++){
            pos[a[i].v] = i;
        }
        bool ok = true;
        for(int i = 1 ; i <= Max ; i ++){
            for(int j = pos[i-1] + 1 ; j <= pos[i] ; j ++){
                while(now <= pos[i] && d[a[now].id] >= k){now ++;}
                if(a[j].v - a[now].v > 1 || now >= pos[i-1]+1){ok = false;break;}
                //刚开始输入的时候node结构体是存储最短距离和节点编号，构造答案时两个值都是节点编号。
                ans.push_back(node(a[now].id , a[j].id));
                d[a[now].id] ++;
                d[a[j].id] ++;
            }
            if(!ok){break;}
            now = max(now,pos[i-1]+1);
        }
        if(!ok || ans.size() == 0){cout << "-1" << endl;}
        else {
            cout << ans.size() << endl;
            for(int i = 0 ; i < ans.size() ; i ++){
                cout << ans[i].v << " " << ans[i].id << endl;
            }
        }
    }
    return 0;
}
```
