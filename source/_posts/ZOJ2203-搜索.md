## 题意
>求一个 1-n 的排列，使得任意相邻 2-d 个数之和均为素数

## 思路
>直接按照题意暴搜即可，注意提前打好素数表，提高判断素数的效率
>注意回溯，搜到答案及时return


```c
#include <bits/stdc++.h>
#include <iostream>
#define inf 0x3f3f3f3f
using namespace std;
typedef long long ll;
typedef pair<int,int> P;
const int maxn = 10000+10;
const int mod = 1e9+7;
const double eps = 1e-10;
int n , m , d;
int prime[maxn];
int vis[maxn];
void init()
{
	prime[0] = prime[1] = 0;
	for(int i = 2 ; i < maxn ; i ++)prime[i] = 1;
	for(int i = 2 ; i < maxn ; i ++){
		if(prime[i]){
			for(int j = i + i ; j < maxn ; j += i)
				prime[j] = 0;
		}
	}
}
int a[maxn] ;
vector<int> ans;
bool check(int pos , int x)
{
	int sum = x;
	for(int i = pos-1 ; i >= 0 && i > pos - d ; i --){
		sum += ans[i];
		if(prime[sum])return false;
	}
	return true;
}
bool ok;
bool dfs(int pos)
{
	if(ans.size() == m-n + 1)return true;
	for(int i = n ; i <= m ; i ++){
		if(!vis[i] && check(pos , i)){
			vis[i] = 1;
			ans.push_back(i);
			if(dfs(pos + 1))return true;
			vis[i] = 0;
			ans.pop_back();
		}
	}
	return false;
}
int main(int argc, char const *argv[])
{
	init();
	while(cin >> n >> m >> d && (n+m+d)){
		ans.clear();
		memset(vis , 0 , sizeof(vis));
		ok = false;
		if(dfs(0)){
			for(int i = 0 ; i <= m-n ; i ++){
				printf("%d%c" , ans[i] , i == m-n ? '\n' : ',');
			}
		}else cout<<"No anti-prime sequence exists."<<endl;
	}
	return 0;
}
```
