# 题目传送门 : [bzoj 1491](http://www.lydsy.com/JudgeOnline/problem.php?id=1491)

>题意：
对于每个节点v定义I(v):<br/>
![](https://odzkskevi.qnssl.com/fa168a3bbb51aebab04cb3969827e47e?v=1531724949)
其中:
* C(s,t)(v)代表经过点v的起点为s终点为t的最短路的数目<br/>
* C(s,t)表示起点为s，终点为t的所有最短路的数目<br/>

>思路：
floyd跑一边最短路，如果GG[i][j] == GG[i][k]+GG[k][j] 即为从i到j的最短路经过点k，<br/>
可以用迪杰斯特拉将最短路径的数目计算出来，经过点k的最短路数量等于num[i][k]*num[k][j];<br/>
Attention： 题目数据会爆int

## Code：
```c
#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <string>
#include <algorithm>
#include <queue>
#include <vector>
#include <time.h>
#include <set>
#define inf 1000000000

using namespace std;
typedef long long ll;
typedef pair<int, int> P;
const int maxn = 1000 + 10;
struct node {
    int to;
    int val;
    node(int tt, int vv): to(tt), val(vv) {}
};
vector<node> G[maxn];
ll GG[maxn][maxn];
ll d[maxn];
ll num[maxn][maxn];
ll cnt[maxn];
int vis[maxn];
int n, m;
void dij(int s) {
    for (int i = 1 ; i <= n ; i ++) {
        d[i] = (i == s ? 0 : inf);
    }
    priority_queue<P , vector<P> , greater<P> > q;
    q.push(P(0, s));
    while (!q.empty()) {
        P now = q.top();
        q.pop();
        int u = now.second;
        if (d[u] < now.first)continue;
        for (int i = 0 ; i < G[u].size() ; i ++) {
            node e = G[u][i];
            if (d[e.to] > d[u] + e.val) {
                d[e.to] = d[u] + e.val;
                q.push(P(d[e.to], e.to));
            }
        }
    }
}
void Cal(int s) {
    memset(cnt, 0, sizeof(cnt));
    memset(vis,0,sizeof(vis));
    cnt[s] = 1;
    priority_queue<P , vector<P> , greater<P> > q;
    q.push(P(0, s));
    while (!q.empty()) {
        P now = q.top();
        q.pop();
        int u = now.second;
        if(vis[u])continue;
        vis[u] = 1;
        for (int i = 0 ; i < G[u].size() ; i ++) {
            node e = G[u][i];
            if (d[e.to] == d[u] + e.val) {
                cnt[e.to] += cnt[u];
                q.push(P(d[e.to], e.to));
            }
        }
    }
}

int main(int argc, char const *argv[])
{
    ios::sync_with_stdio(false);
    while (cin >> n >> m) {
        for (int i = 1 ; i <= n ; i ++) {
            for (int j = 1 ; j <= n ; j ++) {
                GG[i][j] = (i == j ? 0 : inf);
            }
        }
        for(int i = 0 ; i < maxn ; i++){
            G[i].clear();
        }
        for (int i = 0 ; i < m ; i ++) {
            int x, y, z;
            cin >> x >> y >> z;
            G[x].push_back(node(y, z));
            G[y].push_back(node(x, z));
            GG[x][y] = z;
            GG[y][x] = z;
        }
        for (int k = 1 ; k <= n ; k ++) {
            for (int i = 1 ; i <= n ; i ++) {
                for (int j = 1 ; j <= n ; j ++) {
                    GG[i][j] = min(GG[i][j] , GG[i][k] + GG[k][j]);
                }
            }
        }
        for(int i  = 1 ; i <= n ; i ++){
            dij(i);
            Cal(i);
            for(int j = 1 ; j <= n ; j ++){
                if(i == j)continue;
                num[i][j] = cnt[j];
            }
        }
        for (int k = 1 ; k <= n ; k ++) {
            double ans = 0;
            for (int i = 1 ; i <= n ; i ++) {
                for (int j = 1 ; j <= n ; j ++) {
                    if (i == j || i == k || j == k)continue;
                    if (GG[i][j] == GG[i][k] + GG[k][j] ){
                        ans += num[i][k] * num[k][j] * 1.0 / num[i][j];
                    }
                }
            }
            printf("%.3f\n", ans);
        }
    }
    return 0;
}
```
