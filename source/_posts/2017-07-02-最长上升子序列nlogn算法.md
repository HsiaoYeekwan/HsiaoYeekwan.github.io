>用法是：如果是严格LIS，则用的是lower_bound()<br/>
如果是非严格LIS,则用的是upper_bound()<br/>

<!--more-->

```c
int solve(int n){
    memset(book,0,sizeof(book));//book数组为标记数组。
    book[1] = a[1];
    int i,k;
    for(i = 2 ,k = 1 ; i <= n ; i ++){
        if(a[i] >= book[k]){book[++k] = a[i];}
        else{
            int pos = upper_bound(book+1,book+1+k,a[i])-book;
            book[pos] = a[i];
        }
    }
    return k;//k即最长上升子序列的长度
}

```

### Example
题目传送门：[POJ 3670](http://poj.org/problem?id=3670)
```c
#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <string>
#include <algorithm>
#include <queue>
#include <vector>
#include <set>
#define inf 99999999

using namespace std;
typedef long long ll;

const int maxn = 30000+10;
int a[maxn];
int book[maxn];
int solve(int n){
    memset(book,0,sizeof(book));
    book[1] = a[1];
    int i,k;
    for(i = 2 ,k = 1 ; i <= n ; i ++){
        if(a[i] >= book[k]){book[++k] = a[i];}
        else{
            int pos = upper_bound(book+1,book+1+k,a[i])-book;
            book[pos] = a[i];
        }
    }
    return k;
}
int solve2(int n){
    memset(book,0,sizeof(book));
    int i,k;
    book[1] = a[n];
    for(i = n-1 ,k = 1 ; i >= 1 ; i --){
        if(a[i] >= book[k]){book[++k] = a[i];}
        else{
            int pos = upper_bound(book+1,book+1+k,a[i])-book;
            book[pos] = a[i];
        }
    }
    return k;
}

int main(int argc, char const *argv[])
{
    int n;
    //ios::sync_with_stdio(false);
    while(cin >> n){
        for(int i = 1 ; i <= n; i++){
            cin >> a[i];
        }
        int ans1 = n - solve(n);
        //cout << ans1 << endl;
        int ans2 = n - solve2(n);
        //cout << ans1 << ans2 << endl;
        cout << min(ans1,ans2) << endl;
    }
    return 0;
}
```
